## 📋 实施概述

本实施文档详细说明如何实现"内置敏感词扫描规则开关"功能，包括后端配置管理、Worker 逻辑调整、前端 UI 修改和测试验证。

### **核心功能**
1. ✅ 添加 `enableBuiltinRules` 配置项（默认启用）
2. ✅ 禁用时跳过所有内置规则检测（身份证、手机、邮箱等 8 种规则）
3. ✅ 禁用时必须输入有效表达式才能开始扫描
4. ✅ 结果表格动态隐藏所有检测列
5. ✅ 状态栏和工具栏条件显示

### **预计时间**：8-11 小时（1-1.5 个工作日）

---

## 🔧 技术架构

### **数据流**

```
用户操作（设置窗口）
    ↓
更新 config.enableBuiltinRules
    ↓
保存到 config.json
    ↓
开始扫描时传递到 Worker
    ↓
Worker 条件执行检测
    ↓
返回结果（counts 为空或包含数据）
    ↓
前端根据 enableBuiltinRules 动态渲染 UI
```

// 1. 添加当前版本号常量
const CURRENT_CONFIG_VERSION = 2;

// 2. 修改默认配置
const DEFAULT_CONFIG: ScanConfig = {
  // ... 现有默认值
  enabledSensitiveTypes: ['idCard', 'phone', 'email', 'bankCard', 'address', 'ip', 'password'],
  customSensitiveExpression: '',
  
  // 【新增】默认启用内置规则
  enableBuiltinRules: true,
};

// 3. 添加配置迁移函数
function migrateConfig(config: any): ScanConfig {
  // 版本 1 → 版本 2：添加 enableBuiltinRules
  if (!config.version || config.version < CURRENT_CONFIG_VERSION) {
    console.log('[Config] Migrating config from version', config.version || 1, 'to', CURRENT_CONFIG_VERSION);
    
    // 添加 enableBuiltinRules 字段，默认启用
    if (config.enableBuiltinRules === undefined) {
      config.enableBuiltinRules = true;
    }
    
    // 更新版本号
    config.version = CURRENT_CONFIG_VERSION;
    
    // 保存迁移后的配置
    saveConfigSync(config);
  }
  
  return config as ScanConfig;
}

// 4. 在 loadConfig() 中调用迁移函数
export function loadConfig(): ScanConfig {
  try {
    const configPath = getConfigPath();
    if (fs.existsSync(configPath)) {
      const rawData = fs.readFileSync(configPath, 'utf-8');
      const rawConfig = JSON.parse(rawData);
      
      // 【新增】执行配置迁移
      const migratedConfig = migrateConfig(rawConfig);
      
      return validateConfig(migratedConfig);
    }
  } catch (error) {
    console.error('[Config] Failed to load config:', error);
  }
  
  // 返回默认配置（包含 version 字段）
  return { ...DEFAULT_CONFIG, version: CURRENT_CONFIG_VERSION };
}

// 5. 修改 saveConfig() 确保保存 version 字段
export function saveConfig(config: Partial<ScanConfig>): void {
  try {
    const currentConfig = loadConfig();
    const updatedConfig = { 
      ...currentConfig, 
      ...config,
      version: CURRENT_CONFIG_VERSION  // 确保版本号不被覆盖
    };
    
    const configPath = getConfigPath();
    fs.writeFileSync(configPath, JSON.stringify(updatedConfig, null, 2), 'utf-8');
  } catch (error) {
    console.error('[Config] Failed to save config:', error);
  }
}

#### **步骤 1.3：调整 Worker 检测逻辑**

**文件**：`src/workers/file-worker.ts`

```typescript
// 1. 接收新参数
parentPort?.on('message', async (task: {
  taskId: string;
  filePath: string;
  enableBuiltinRules?: boolean;  // 【新增】
  enabledSensitiveTypes: string[];
  customExpression?: string;
  // ... 其他参数
}) => {
  const {
    taskId,
    filePath,
    enableBuiltinRules = true,  // 【新增】默认启用
    enabledSensitiveTypes,
    customExpression,
    // ...
  } = task;
  
  try {
    // 2. 读取文件内容（已有逻辑）
    const text = await readFileContent(filePath);
    
    // 3. 条件执行内置规则检测
    let counts: Record<string, number> = {};
    let totalCount = 0;
    
    if (enableBuiltinRules) {
      // ✅ 执行所有内置规则检测
      const result = detectSensitiveData(text, enabledSensitiveTypes);
      counts = result.counts;      // { idCard: 2, phone: 5, email: 3, ... }
      totalCount = result.total;   // 10
    } else {
      // ❌ 跳过所有内置规则检测
      counts = {};                 // 空对象
      totalCount = 0;              // 总数为 0
    }
    
    // 4. 始终执行自定义表达式评估（无论 enableBuiltinRules 的值）
    let expressionMatched = 0;
    if (customExpression && customExpression.trim()) {
      try {
        const isMatched = evaluateCustomExpression(text, customExpression);
        if (isMatched) {
          expressionMatched = 1;
        }
      } catch (error) {
        console.error(`[Worker] Expression evaluation failed for ${filePath}:`, error);
      }
    }
    
    // 5. 返回结果
    parentPort?.postMessage({
      type: 'scanResult',
      taskId,
      filePath,
      counts,                      // 禁用时为 {}
      total: totalCount,           // 禁用时为 0
      expressionMatched,           // 可能为 0 或 1
      fileSize: stats.size,
      modifiedTime: stats.mtimeMs,
    });
    
  } catch (error) {
    parentPort?.postMessage({
      type: 'scanError',
      taskId,
      filePath,
      error: error.message,
    });
  }
});
```

    const mockDetectSensitiveData = vi.fn(() => ({
      counts: { idCard: 2, phone: 3 },
      total: 5
    }));
    
    const enableBuiltinRules = true;
    let counts: Record<string, number> = {};
    let totalCount = 0;
    
    if (enableBuiltinRules) {
      const result = mockDetectSensitiveData(text, enabledSensitiveTypes);
      counts = result.counts;
      totalCount = result.total;
    }
    
    expect(counts).toEqual({ idCard: 2, phone: 3 });
    expect(totalCount).toBe(5);
    expect(mockDetectSensitiveData).toHaveBeenCalled();

    const mockEvaluateExpression = vi.fn(() => true);

    let expressionMatched = 0;
    if (mockEvaluateExpression()) {
      expressionMatched = 1;

    // 场景 2：禁用内置规则
    expressionMatched = 0;
    if (mockEvaluateExpression()) {
      expressionMatched = 1;
    }
    expect(expressionMatched).toBe(1);
    
    expect(mockEvaluateExpression).toHaveBeenCalledTimes(2);
  });
});

  // 如果启用了内置规则，始终可以扫描
  if (config.value.enableBuiltinRules !== false) {
    return true;
  }
  
  // 如果禁用了内置规则，必须有有效的自定义表达式
  if (!customExpression.value.trim()) {
    return false;  // 表达式为空
  }
  
  // 检查表达式语法是否正确
  if (expressionError.value) {
    return false;  // 表达式有语法错误
  }
  
  return true;
});

// 【新增】获取禁用原因
const getDisableReason = () => {
  if (isScanning.value) {
    return '正在扫描中...';
  }
  
  if (appStore.selectedPaths.length === 0) {
    return '请先选择扫描路径';
  }
  
  if (config.value.enableBuiltinRules === false) {
    if (!customExpression.value.trim()) {
      return '请输入自定义表达式';
    }
    if (expressionError.value) {
      return '表达式语法错误';
    }
  }
  
  return '';
};

// 【新增】验证表达式（已有逻辑，确保存在）
const validateExpression = () => {
  if (!customExpression.value.trim()) {
    expressionError.value = null;
    return;
  }
  
  try {
    // 调用表达式解析器验证
    const result = parseExpression(customExpression.value);
    expressionError.value = result.valid ? null : result.error;
  } catch (error) {
    expressionError.value = '表达式语法错误';
  }
};

// 【修改】开始扫描
const handleStartScan = async () => {
  if (!canStartScan.value) {
    return;
  }
  
  isScanning.value = true;
  scanProgress.value = 0;
  scannedFiles.value = 0;
  totalFiles.value = 0;
  sensitiveFilesCount.value = 0;
  totalSensitiveItems.value = 0;
  scanDuration.value = 0;
  
  const startTime = Date.now();
  
  try {
    // 构建扫描配置
    const scanConfig = {
      enableBuiltinRules: config.value.enableBuiltinRules,  // 【新增】
      enabledSensitiveTypes: config.value.enableBuiltinRules 
        ? [...config.value.enabledSensitiveTypes] 
        : [],  // 禁用时传空数组
      customSensitiveExpression: customExpression.value.trim() || undefined,
      // ... 其他配置
    };
    
    // 调用后端扫描 API
    await electronAPI.startScan(appStore.selectedPaths, scanConfig);
    
    // 监听扫描进度（已有逻辑）
    electronAPI.onScanProgress((progress) => {
      scanProgress.value = progress.percent;
      scannedFiles.value = progress.scanned;
      totalFiles.value = progress.total;
    });
    
    // 监听扫描结果（已有逻辑）
    electronAPI.onScanResult((result) => {
      appStore.addScanResult(result);
      
      // 统计敏感文件和信息
      if (config.value.enableBuiltinRules !== false) {
        if (result.total > 0) {
          sensitiveFilesCount.value++;
          totalSensitiveItems.value += result.total;
        }
      }
    });
    
    // 等待扫描完成
    await waitForScanComplete();
    
  } catch (error) {
    console.error('[App] Scan failed:', error);
    alert('扫描失败：' + error.message);
  } finally {
    isScanning.value = false;
    scanDuration.value = Date.now() - startTime;
  }
};

// 【修改】批量删除
const handleBatchDelete = async () => {
  if (selectedRows.value.length === 0) {
    return;
  }
  
  const confirmed = confirm(`确定要删除选中的 ${selectedRows.value.length} 个文件吗？`);
  if (!confirmed) {
    return;
  }
  
  try {
    for (const filePath of selectedRows.value) {
      await electronAPI.deleteFile(filePath);
      appStore.removeScanResult(filePath);
    }
    
    selectedRows.value = [];
    alert('批量删除成功');
  } catch (error) {
    console.error('[App] Batch delete failed:', error);
    alert('批量删除失败：' + error.message);
  }
};

// 【新增】监听表达式变化，实时验证
watch(customExpression, debounce((val) => {
  validateExpression();
}, 500));
</script>
