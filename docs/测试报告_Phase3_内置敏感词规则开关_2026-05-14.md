```typescript
// App.vue - isStartScanDisabled 计算属性
const isStartScanDisabled = computed(() => {
  if (isScanning.value || isCancelling.value) return true
  if (appStore.selectedPaths.size === 0) return true
  
  // 如果禁用内置规则，必须有有效的表达式
  if (config.value.enableBuiltinRules === false) {
    const expr = searchExpression.value.trim()
    if (!expr) return true  // ✅ 空表达式时禁用按钮
    if (expressionValidationError.value) return true
  }
  
  return false
})
```
